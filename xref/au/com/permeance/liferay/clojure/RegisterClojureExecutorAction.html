17  package au.com.permeance.liferay.clojure;
18  
19  import com.liferay.portal.kernel.events.SimpleAction;
20  import com.liferay.portal.kernel.scripting.Scripting;
21  import com.liferay.portal.kernel.scripting.ScriptingExecutor;
22  import com.liferay.portal.kernel.scripting.ScriptingUtil;
23  
24  /**
25   * This class provides a liferay action which registers the ClojureScriptingExecutor to liferay for script execution.
26   *
27   * @see SimpleAction
28   * @see ClojureScriptingExecutor
29   */
30  public class RegisterClojureExecutorAction extends SimpleAction {
31  
32      /**
33       * Executes the action of creating and registering a {@link ClojureScriptingExecutor}.
34       *
35       * @param ids the company IDs for the portal (not used)
36       */
37      @Override
38      public final void run(final String[] ids) {
39          final ScriptingExecutor executor = createClojureScriptingExecutor();
40  
41          registerScriptingExecutor(executor);
42      }
43  
44      /**
45       * Creates a {@link ClojureScriptingExecutor} which is associated with the {@link ClassLoader} of the current
46       * thread (the hook's {@link ClassLoader} rather than the Portal's under which scripts are invoked).
47       *
48       * @return the newly created {@link ScriptingExecutor}.
49       */
50      protected final ScriptingExecutor createClojureScriptingExecutor() {
51          @SuppressWarnings("PMD.DoNotUseThreads")
52          final Thread thread = Thread.currentThread();
53          final ClassLoader classLoader = thread.getContextClassLoader();
54  
55          return new ClojureScriptingExecutor(classLoader);
56      }
57  
58      /**
59       * Registers the supplied {@link ScriptingExecutor} with liferay by calling {@link ScriptingUtil#getScripting()} to
60       * obtain the scripting engine, and then {@link Scripting#addScriptionExecutor(String, ScriptingExecutor)}.
61       *
62       * @param executor the {@link ScriptingExecutor} to register.
63       */
64      protected final void registerScriptingExecutor(final ScriptingExecutor executor) {
65          final String language = executor.getLanguage();
66          final Scripting scripting = ScriptingUtil.getScripting();
67  
68          scripting.addScriptionExecutor(language, executor);
69      }
70  
71  }
