017    package au.com.permeance.liferay.clojure;
018    
019    import com.liferay.portal.kernel.scripting.ExecutionException;
020    import com.liferay.portal.kernel.scripting.ScriptingException;
021    import org.junit.Test;
022    
023    import java.util.Collections;
024    import java.util.Map;
025    import java.util.concurrent.Callable;
026    
027    import static com.liferay.portal.kernel.util.StringPool.BLANK;
028    import static java.util.Collections.singleton;
029    import static org.fest.assertions.Assertions.assertThat;
030    import static org.fest.assertions.MapAssert.entry;
031    
032    public class ClojureScriptingExecutorTest {
033    
034        @Test
035        public void testEval() throws Exception {
036            ClojureScriptingExecutor executor = new ClojureScriptingExecutor();
037    
038            Map<String, Object> result = executor.eval(Collections.<String>emptySet(),
039                                                       Collections.<String, Object>singletonMap("one", 1),
040                                                       Collections.<String>singleton("result"),
041                                                       "(let [one (get input-objects \"one\")] {\"result\" (+ one one)})");
042    
043            assertThat(result)
044                    .hasSize(1)
045                    .includes(entry("result", 2L));
046        }
047    
048        @Test(expected = ExecutionException.class)
049        public void testEvalWithAllowedClasses() throws Exception {
050            ClojureScriptingExecutor executor = new ClojureScriptingExecutor();
051    
052            executor.eval(singleton("class"),
053                          Collections.<String, Object>emptyMap(),
054                          Collections.<String>emptySet(),
055                          BLANK);
056        }
057    
058        @Test
059        public void testGetLanguage() throws Exception {
060            ClojureScriptingExecutor executor = new ClojureScriptingExecutor();
061    
062            String language = executor.getLanguage();
063    
064            assertThat(language).isEqualTo("clojure");
065        }
066    
067    
068        @Test
069        public void testCreateCallableForScript() throws Exception {
070            ClojureScriptingExecutor executor = new ClojureScriptingExecutor();
071    
072            Callable<Map<String, Object>> callable = executor.createCallableForScript(Collections.<String, Object>singletonMap("one", 1),
073                                                                                      Collections.<String>singleton("result"),
074                                                                                      "(let [one (get input-objects \"one\")] {\"result\" (+ one one)})");
075            Map<String, Object> result = callable.call();
076    
077            assertThat(callable).isInstanceOf(CallUnderClassLoader.class);
078            assertThat(result)
079                    .hasSize(1)
080                    .includes(entry("result", 2L));
081        }
082    
083        @Test(expected = ScriptingException.class)
084        public void testCallAndWrapException() throws Exception {
085            ClojureScriptingExecutor executor = new ClojureScriptingExecutor();
086    
087            executor.callAndWrapException(new Callable<Object>() {
088                @Override
089                public Object call() throws Exception {
090                    throw new Exception();
091                }
092            });
093        }
094    
095    }
