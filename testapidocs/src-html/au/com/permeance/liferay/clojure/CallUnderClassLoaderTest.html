017    package au.com.permeance.liferay.clojure;
018    
019    import org.junit.Test;
020    import org.junit.runner.RunWith;
021    import org.mockito.Mock;
022    import org.mockito.runners.MockitoJUnitRunner;
023    
024    import java.util.concurrent.Callable;
025    
026    import static org.fest.assertions.Assertions.assertThat;
027    
028    @RunWith(MockitoJUnitRunner.class)
029    public class CallUnderClassLoaderTest {
030    
031        @Mock
032        private ClassLoader mockClassLoader;
033    
034        @Test
035        public void testCall() throws Exception {
036            MockCallable mockCallable = new MockCallable();
037            CallUnderClassLoader<Object> callUnderClassLoader = new CallUnderClassLoader<Object>(mockCallable, mockClassLoader);
038            ClassLoader originalClassLoader = Thread.currentThread().getContextClassLoader();
039    
040            callUnderClassLoader.call();
041    
042            assertThat(Thread.currentThread().getContextClassLoader()).isSameAs(originalClassLoader);
043            assertThat(mockCallable.callingClassLoader).isSameAs(mockClassLoader);
044        }
045    
046        private class MockCallable implements Callable<Object> {
047    
048            private ClassLoader callingClassLoader;
049    
050            @Override
051            public Object call() throws Exception {
052                callingClassLoader = Thread.currentThread().getContextClassLoader();
053                return null;
054            }
055    
056        }
057    
058    }
