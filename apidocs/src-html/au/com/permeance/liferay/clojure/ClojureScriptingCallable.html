017    package au.com.permeance.liferay.clojure;
018    
019    import com.liferay.portal.kernel.log.Log;
020    import com.liferay.portal.kernel.log.LogFactoryUtil;
021    
022    import java.util.Collection;
023    import java.util.Map;
024    import java.util.Set;
025    import java.util.concurrent.Callable;
026    import java.util.concurrent.ConcurrentHashMap;
027    import java.util.concurrent.ConcurrentMap;
028    
029    import static java.lang.String.format;
030    import static java.util.Collections.emptyMap;
031    
032    /**
033     * This class wraps the clojure generated ClojureScriptable protocol and ClojureScriptableImpl implementation classes
034     * for execution of scripts from within Liferay.
035     */
036    public class ClojureScriptingCallable implements Callable<Map<String, Object>> {
037    
038        /**
039         * Stores the logger for reporting errors.
040         */
041        private static final Log LOG = LogFactoryUtil.getLog(ClojureScriptingCallable.class);
042    
043        /**
044         * Stores a reference to an object implementing the ClojureScriptable protocol for executing scripts.
045         */
046        private static final ClojureScriptable CLOJURE_SCRIPTABLE = new ClojureScriptableImpl();
047    
048        /**
049         * Stores the map of names to values of input objects, as supplied through
050         * {@link com.liferay.portal.kernel.scripting.BaseScriptingExecutor#eval(Set, Map, Set, String)}.
051         */
052        private final transient Map<String, Object> inputObjects;
053    
054        /**
055         * Stores the map of names to values of output names, as supplied through
056         * {@link com.liferay.portal.kernel.scripting.BaseScriptingExecutor#eval(Set, Map, Set, String)}.
057         */
058        private final transient Set<String> outputNames;
059    
060        /**
061         * Stores the script to execute, as supplied through
062         * {@link com.liferay.portal.kernel.scripting.BaseScriptingExecutor#eval(Set, Map, Set, String)}.
063         */
064        private final transient String script;
065    
066        /**
067         * Creates a new instance of ClojureScriptingCallable, storing the supplied input objects, output names and script
068         * for use in the {@link #call()} method.
069         *
070         * @param inputObjects a map of input names to values to expose to the script.
071         * @param outputNames  a set of strings representing keys of the result map to return from {@link #call()}.
072         * @param script       the script to execute.
073         */
074        public ClojureScriptingCallable(final Map<String, Object> inputObjects,
075                                        final Set<String> outputNames,
076                                        final String script) {
077            this.inputObjects = inputObjects;
078            this.outputNames = outputNames;
079            this.script = script;
080        }
081    
082        /**
083         * Executes the {@link #script} associated with this instance, supplying the input objects and output names to
084         * the {@link #CLOJURE_SCRIPTABLE} to pass onto the script.
085         *
086         * @return a map representing the result of calling the script and extracting the {@link #outputNames} from its
087         *         result. Note: there are no guarantees made that this map will contain any of the requested keys - that's
088         *         up to the individual script to provide. However this method does ensure that any map keys which aren't
089         *         in {@link #outputNames} are removed.
090         */
091        @Override
092        @SuppressWarnings("unchecked")
093        public final Map<String, Object> call() {
094            final Object result = CLOJURE_SCRIPTABLE.run_script(inputObjects, outputNames, script);
095    
096            if (outputNames == null
097                || outputNames.isEmpty()
098                || result == null) {
099                return emptyMap();
100            }
101    
102            if (!(result instanceof Map)) {
103                LOG.warn(format("Unknown return type from clojure script: %s (%s)", result, result.getClass()));
104                return emptyMap();
105            }
106    
107            return retainKeys((Map<String, Object>) result, outputNames);
108        }
109    
110        /**
111         * This method retains the supplied keys in the supplied map - keys in the map which are missing from the set are
112         * absent from the returned map.
113         *
114         * @param map  the map to retain keys within.
115         * @param keys the keys to retain.
116         * @param <K>  the type of keys maintained by this map
117         * @param <V>  the type of mapped values
118         *
119         * @return a new map containing the keys supplied.
120         */
121        protected final <K, V> Map<K, V> retainKeys(final Map<K, V> map,
122                                                    final Collection<K> keys) {
123            final ConcurrentMap<K, V> result = new ConcurrentHashMap<K, V>(keys.size());
124    
125            for (final K key : keys) {
126                final V value = map.get(key);
127                if (value != null) {
128                    result.put(key, value);
129                }
130            }
131    
132            return result;
133        }
134    
135    }
