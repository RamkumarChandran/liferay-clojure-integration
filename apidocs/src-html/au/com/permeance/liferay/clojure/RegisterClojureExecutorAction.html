017    package au.com.permeance.liferay.clojure;
018    
019    import com.liferay.portal.kernel.events.SimpleAction;
020    import com.liferay.portal.kernel.scripting.Scripting;
021    import com.liferay.portal.kernel.scripting.ScriptingExecutor;
022    import com.liferay.portal.kernel.scripting.ScriptingUtil;
023    
024    /**
025     * This class provides a liferay action which registers the ClojureScriptingExecutor to liferay for script execution.
026     *
027     * @see SimpleAction
028     * @see ClojureScriptingExecutor
029     */
030    public class RegisterClojureExecutorAction extends SimpleAction {
031    
032        /**
033         * Executes the action of creating and registering a {@link ClojureScriptingExecutor}.
034         *
035         * @param ids the company IDs for the portal (not used)
036         */
037        @Override
038        public final void run(final String[] ids) {
039            final ScriptingExecutor executor = createClojureScriptingExecutor();
040    
041            registerScriptingExecutor(executor);
042        }
043    
044        /**
045         * Creates a {@link ClojureScriptingExecutor} which is associated with the {@link ClassLoader} of the current
046         * thread (the hook's {@link ClassLoader} rather than the Portal's under which scripts are invoked).
047         *
048         * @return the newly created {@link ScriptingExecutor}.
049         */
050        protected final ScriptingExecutor createClojureScriptingExecutor() {
051            @SuppressWarnings("PMD.DoNotUseThreads")
052            final Thread thread = Thread.currentThread();
053            final ClassLoader classLoader = thread.getContextClassLoader();
054    
055            return new ClojureScriptingExecutor(classLoader);
056        }
057    
058        /**
059         * Registers the supplied {@link ScriptingExecutor} with liferay by calling {@link ScriptingUtil#getScripting()} to
060         * obtain the scripting engine, and then {@link Scripting#addScriptionExecutor(String, ScriptingExecutor)}.
061         *
062         * @param executor the {@link ScriptingExecutor} to register.
063         */
064        protected final void registerScriptingExecutor(final ScriptingExecutor executor) {
065            final String language = executor.getLanguage();
066            final Scripting scripting = ScriptingUtil.getScripting();
067    
068            scripting.addScriptionExecutor(language, executor);
069        }
070    
071    }
